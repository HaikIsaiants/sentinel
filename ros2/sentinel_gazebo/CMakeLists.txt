cmake_minimum_required(VERSION 3.16)

project(sentinel_gazebo LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

find_package(ament_cmake REQUIRED)
find_package(Protobuf REQUIRED)
find_package(rclcpp REQUIRED)
find_package(ros_gz_interfaces REQUIRED)

get_filename_component(SENTINEL_ROOT "${CMAKE_CURRENT_SOURCE_DIR}/../.." ABSOLUTE)
set(SENTINEL_GENERATED "${CMAKE_CURRENT_BINARY_DIR}/generated")
set(SENTINEL_PROTO "${SENTINEL_ROOT}/proto/sentinel/v1/sentinel.proto")
set(SENTINEL_PROTO_SOURCE "${SENTINEL_GENERATED}/sentinel/v1/sentinel.pb.cc")
set(SENTINEL_PROTO_HEADER "${SENTINEL_GENERATED}/sentinel/v1/sentinel.pb.h")

# TODO: Can an overlay reuse the bindings generated by the root build?
file(MAKE_DIRECTORY "${SENTINEL_GENERATED}")

add_custom_command(
    OUTPUT "${SENTINEL_PROTO_SOURCE}" "${SENTINEL_PROTO_HEADER}"
    COMMAND "${Protobuf_PROTOC_EXECUTABLE}"
        "--proto_path=${SENTINEL_ROOT}/proto"
        "--cpp_out=${SENTINEL_GENERATED}"
        "${SENTINEL_PROTO}"
    DEPENDS "${SENTINEL_PROTO}"
    VERBATIM
)

add_executable(sentinel_gazebo_adapter
    src/adapter.cpp
    "${SENTINEL_PROTO_SOURCE}"
    "${SENTINEL_ROOT}/proto/src/framing.cpp"
)

target_include_directories(sentinel_gazebo_adapter PRIVATE
    "${SENTINEL_GENERATED}"
    "${SENTINEL_ROOT}/proto/include"
)

target_link_libraries(sentinel_gazebo_adapter protobuf::libprotobuf)
ament_target_dependencies(sentinel_gazebo_adapter rclcpp ros_gz_interfaces)
target_compile_options(sentinel_gazebo_adapter PRIVATE -Wall -Wextra -Wpedantic)

install(TARGETS sentinel_gazebo_adapter DESTINATION lib/${PROJECT_NAME})
install(DIRECTORY launch worlds DESTINATION share/${PROJECT_NAME})

ament_package()
